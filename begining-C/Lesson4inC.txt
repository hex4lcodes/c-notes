#include <stdio.h>

int main(void) {
    
    //defining variables
    float miles;
    float kilometres; 

    //getting the kilometres - dont forget newline (\n)
    printf("Enter a distance in kilometres: \n");
    scanf("%f", &kilometres); //dont forget the & in front of variables in scanf or you'll bug out

    //conversion and printing
    miles = kilometres * 0.621371;

    printf("%.4f km is %.4f miles \n", kilometres, miles);
    return 0;
}
