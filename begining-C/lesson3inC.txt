//Simple Calculator - no if statements

#include <stdio.h>

int main() {

//numbers
int num1;
int num2;

num1 = 15;
num2 = 4;

//results
int result_int;
float result_float;

//dont forget your new line characters, C isnt as focused as Python, they will print on same line w/o them

//addition
result_int = num1 + num2;
printf("The sum of %d and %d is %d \n", num1, num2, result_int);

//subtraction
result_int = num1 - num2;
printf("The difference of %d and %d is %d \n", num1, num2, result_int);

//product
result_int = num1 * num2;
printf("The product of %d and %d is %d \n", num1, num2, result_int);


//int division
result_int = num1 / num2;
printf("The difference of %d and %d is %d \n", num1, num2, result_int);

//float division
float fnum1 = 15.0; //made a new variable for the converted type of num1 for accurate division

result_float = fnum1 / num2;
printf("The REAL difference of %.1f and %d is %.4f \n", fnum1, num2, result_float);  //round ans to 4 decimal places

//modulo (remainder)
result_int = num1 % num2;
printf("The remainder of %d and %d is %d \n", num1, num2, result_int);
return 0;
}



//Incrementing / De-incrementing

#include <stdio.h>

int main() {

//declaring values
int val1 = 10;
int val2 = 20;

int output1;
int output2;

//output creation
output1 = val1++;
printf("now %d = %d \n", output1, val1); //have to print them backwards to get original values

output2 = val2--;
printf("now %d = %d \n", output2, val2);
return 0;
}
